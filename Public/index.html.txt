      <input id="projName" placeholder="Project Name">

<script>
let token = "";
let userRole = "";

/* AUTH */
async function register(){
  await fetch("/api/auth/register",{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({
      name:name.value,
      email:email.value,
      password:password.value,
      role:role.value
    })
  });
  alert("Registered");
}

async function login(){
  const res = await fetch("/api/auth/login",{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({
      email:email.value,
      password:password.value
    })
  });
  const data = await res.json();
  token = data.token;
  userRole = data.role;

  alert("Login success");

  loadProjects();
  loadUsers();
  loadTasks();
}

/* PROJECT */
async function createProject(){
  if(userRole !== "Admin"){
    return alert("Only Admin can create projects");
  }

  await fetch("/api/projects",{
    method:"POST",
    headers:{
      "Content-Type":"application/json",
      "Authorization":token
    },
    body:JSON.stringify({
      name:projName.value,
      description:projDesc.value
    })
  });

  loadProjects();
}

async function loadProjects(){
  const res = await fetch("/api/projects",{ headers:{Authorization:token}});
  const data = await res.json();

  projectList.innerHTML="";
  data.forEach(p=>{
    let opt=document.createElement("option");
    opt.value=p._id;
    opt.text=p.name;
    projectList.appendChild(opt);
  });
}

/* USERS */
async function loadUsers(){
  const res = await fetch("/api/auth/users",{ headers:{Authorization:token}});
  const users = await res.json();

  userList.innerHTML="";
  users.forEach(u=>{
    let opt=document.createElement("option");
    opt.value=u._id;
    opt.text=u.name;
    userList.appendChild(opt);
  });
}

/* TASK */
async function createTask(){
  await fetch("/api/tasks",{
    method:"POST",
    headers:{
      "Content-Type":"application/json",
      "Authorization":token
    },
    body:JSON.stringify({
      title:taskTitle.value,
      project:projectList.value,
      assignedTo:userList.value,
      dueDate:dueDate.value
    })
  });
  loadTasks();
}

async function loadTasks(){
  const res = await fetch("/api/tasks",{ headers:{Authorization:token}});
  const tasks = await res.json();

  tasksDiv.innerHTML="";

  let total=0, done=0, pending=0;

  tasks.forEach(t=>{
    total++;
    if(t.status==="Done") done++;
    else pending++;

    let overdue = t.dueDate && new Date(t.dueDate) < new Date() && t.status!=="Done";

    let div=document.createElement("div");
    div.className="task";

    div.innerHTML=`
      <b>${t.title}</b> 
      <span class="${t.status==='Done'?'done':'pending'}">(${t.status})</span>
      ${overdue ? "<span class='overdue'>⚠ Overdue</span>" : ""}
      <br>Assigned: ${t.assignedTo?.name || "N/A"}
      <br>Due: ${t.dueDate ? t.dueDate.substring(0,10):"N/A"}
      <br>
      <button onclick="updateStatus('${t._id}','Done')">Done</button>
      <button onclick="updateStatus('${t._id}','In Progress')">Progress</button>
    `;

    tasksDiv.appendChild(div);
  });

  totalSpan.innerText=total;
  doneSpan.innerText=done;
  pendingSpan.innerText=pending;
}

/* UPDATE */
async function updateStatus(id,status){
  await fetch("/api/tasks/"+id,{
    method:"PUT",
    headers:{
      "Content-Type":"application/json",
      "Authorization":token
    },
    body:JSON.stringify({status})
  });
  loadTasks();
}
